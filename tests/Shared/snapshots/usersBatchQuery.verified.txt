query all_users($site:String!, $tenant:String!, $userIds:[String!]){
    base(condition:{site:$site, tenant_id:$tenant, is_test:false}, filters:{user_id:{in:$userIds}}){
        edges{
            node{
                UserName:user_name
                Balance:user_balance
                Email:user_email
                UserId:user_id
            }
        }
    }
}
